int device1 = 0;

#define esp_baudrate 115200
iotgecko gecko = iotgecko(esp_baudrate);

bool notConected = true;
bool login = false;

String id = "thomasikenna@gmail.com";//iotgecko login id
String pass = "8344";//iotgecko login password
String ssid = "iot";//Wifi SSID
String pass_key = "project1234";//Wifi password   

const int no_of_data = 5;
String patient_data[no_of_data];
LiquidCrystal lcd(A0, A1, A2, A3, A4, A5); //RS, EN, D4, D5, D6, D7
void setup() {
  // put your setup code here, to run once:
Serial.begin(115200);
  Serial.println(F("RF based hospital asset"));
  lcd.begin(16, 2);
Serial.begin(9600);
pinMode(Add1,OUTPUT);
pinMode(Add0, OUTPUT );
pinMode(VT, INPUT );
pinMode(data0, INPUT );
pinMode(data1, INPUT );
pinMode(data2, INPUT );
pinMode(data3, INPUT );
  delay(500);
  lcd.clear();
  lcd.setCursor(1,0);
  lcd.print("Connecting to");
  lcd.setCursor(4,1);
  lcd.print(" Wifi");
  delay(1000);
  Serial.println(F("conncecting to wifi"));
  Serial.print(ssid);
  Serial.print("\t");
  Serial.println(pass_key);  
  while(notConected)
  {
    if(gecko.GeckoConnect(ssid,pass_key))//connect to wifi with given SSID and password
    {
      Serial.println(F("connceted to wifi..."));
      lcd.clear();
      lcd.setCursor(0,0);
      lcd.print("Wifi connected");
      delay(2000);
      notConected = false;
    }
    else
    {
      Serial.println(F("can't connect to wifi")); 
      lcd.clear();
      lcd.setCursor(0,0);
      lcd.print("Can't Connect");
      lcd.setCursor(1,1);
      lcd.print("to Wifi");
      delay(2000);
    }
    delay(1000);
  }
  Serial.println(F("connecting to iotgecko.com"));
  while(!login)
  {
    if(gecko.GeckoVerify(id,pass))//login to iotgecko.com with given ID and password
    {
      Serial.println(F("connected succesfully"));
      lcd.clear();
      lcd.setCursor(0,0);
      lcd.print("Connected to");
      lcd.setCursor(1,1);
      lcd.print(" iotgecko");
         delay(2000);
      login = true;
    }
    else
    {
      Serial.println(F("fail to connect"));
      lcd.clear();
      lcd.setCursor(0,0);
      lcd.print("connection to");
      lcd.setCursor(0,1);
      lcd.print("iotgecko failed");
         delay(2000);
    }
  }


}

void loop()
{
  // put your main code here, to run repeatedly:
  lcd.clear();
  lcd.setCursor(3,0);
  lcd.print("RF hospital");
  lcd.setCursor(5,1);
  lcd.print("Asset");
  delay(2000); // delay 2 secs
  while(1)
  {
    scan();
    patient_data[0] = (String)1; // room 1
    patient_data[1] = (String)device1;
    patient_data[2] = (String)device2;
    patient_data[3] = (String)device3;
    patient_data[4] = (String)device4;
    lcd.clear();
    if (device1==1)
    {
     lcd.setCursor(0,0); // row, coloumn
     lcd.print("Device1");
    }
    if (device2==1)
    {
     lcd.setCursor(9,0); // row, coloumn
     lcd.print("Device2");
    }
    if (device3==1)
    {
     lcd.setCursor(0,1); // row, coloumn
     lcd.print("Device3");
    }
    if (device4==1)
    {
     lcd.setCursor(9,1); // row, coloumn
     lcd.print("Device4");
    }
    if((device1!=1) && (device2!=1) && (device3!=1) && (device4!=1))
    {
      lcd.clear();
      lcd.print("No devices"); 
      lcd.setCursor(0,1);
      lcd.print("found..");
    }   
  int iot_status = gecko.SendGParams(patient_data,no_of_data);//send data to iotgecko.com
  if(iot_status == InvalidUserIdOrPassword)
  {
    Serial.println(F("Invalid UserId or Password"));
    while(1);
  }
  else if(iot_status == InvalidData)
  {
    Serial.println(F("Invalid Data"));
    while(1);
  }
  else if(iot_status == HitLimitExceed)
  {
    Serial.println(F("Limit Exceed"));
  }
  else if(iot_status == VALID)
  {
    Serial.println(F("data send succesfully....:)"));
  }
  else
  {
    Serial.println(F("connection lost"));
    while(!gecko.GeckoReconnect())//reconnect to the iotgecko.com 
    {
      Serial.println(F("conncetion failed.....reconnecting"));
      delay(2000);
    }
    Serial.println(F("connected succesfully"));
   delay(1000);  
  }
}
}

void scan()
{
  lcd.clear();
  lcd.setCursor(2,0);
  lcd.print("Scanning for");
  lcd.setCursor(4,1);
  lcd.print("Devices.");
  digitalWrite(Add0, LOW);
  digitalWrite(Add1, LOW);
  t1 = millis();
  while(1)
  {
    if ((digitalRead(VT)==HIGH)&&(digitalRead(data0)==HIGH)&&(digitalRead(data1)==LOW)&&(digitalRead(data2)==LOW)&&(digitalRead(data3)==LOW))
    {
      device1 = 1;
//      lcd.clear();
//      lcd.print("Device1");
      break;

    if((millis() - t1) >= 8000)

//      lcd.print("Device3");
